Toán tử & Biểu thức

Tất cả toán tử trong Python

# Toán tử số học
a = 10
b = 3

print("a + b =", a + b)    # 13 - Cộng
print("a - b =", a - b)    # 7 - Trừ
print("a * b =", a * b)    # 30 - Nhân
print("a / b =", a / b)    # 3.333... - Chia
print("a // b =", a // b)   # 3 - Chia nguyên
print("a % b =", a % b)    # 1 - Chia dư
print("a ** b =", a ** b)   # 1000 - Lũy thừa

# Toán tử so sánh
print("a == b:", a == b)    # False - Bằng
print("a != b:", a != b)    # True - Khác
print("a > b:", a > b)     # True - Lớn hơn
print("a < b:", a < b)     # False - Nhỏ hơn
print("a >= b:", a >= b)   # True - Lớn hơn hoặc bằng

# Toán tử logic
x = True
y = False

print("x and y:", x and y)  # False - VÀ
print("x or y:", x or y)    # True - HOẶC
print("not x:", not x)       # False - PHỦ ĐỊNH

# Toán tử gán
c = 5
c += 3   # c = c + 3 → 8
c -= 2   # c = c - 2 → 6
c *= 2   # c = c * 2 → 12
c /= 3   # c = c / 3 → 4.0

Toán tử Python:

  • + - * /: Toán tử số học cơ bản
  • // % **: Chia nguyên, chia dư, lũy thừa
  • == != > < >= <=: Toán tử so sánh
  • and or not: Toán tử logic
  • += -= *= /=: Toán tử gán kết hợp

Câu lệnh điều kiện

if, elif, else đầy đủ

# if cơ bản

    print("Bạn đã trưởng thành")
else:
    print("Bạn chưa trưởng thành")

# if-elif-else nhiều điều kiện

    print("Xuất sắc! Điểm A")

    print("Giỏi! Điểm B")

    print("Khá! Điểm C")
elif score >= 50:
    print("Trung bình! Điểm D")
else:
    print("Yếu! Điểm F")

# Điều kiện lồng nhau

    if has_license:
        print("Bạn được lái xe")
    else:
        print("Bạn cần bằng lái")
else:
    print("Bạn chưa đủ tuổi lái xe")

# Toán tử ternary (if ngắn gọn)

status = "Trưởng thành" if age >= 18 else "Vị thành niên"
print(f"Tuổi {age}: {status}")

Giải thích:

  • if: Kiểm tra điều kiện đầu tiên
  • elif: Kiểm tra điều kiện tiếp theo nếu if sai
  • else: Thực hiện khi tất cả điều kiện sai
  • Lồng if: if bên trong if khác
  • Ternary: Cú pháp ngắn gọn: x if condition else y

Vòng lặp for

for loop với nhiều ví dụ

# Lặp qua list
fruits = ["táo", "chuối", "cam", "nho"]

for fruit in fruits:
    print(f"Tôi thích ăn {fruit}")

# Lặp với range()
print("Đếm từ 0 đến 4:")
for i in range(5):          # 0, 1, 2, 3, 4
    print(i)

print("Đếm từ 1 đến 10:")
for i in range(1, 11):    # 1, 2, 3, ..., 10
    print(i)

print("Số chẵn từ 2 đến 10:")
for i in range(2, 11, 2):  # 2, 4, 6, 8, 10
    print(i)

# Lặp qua string
word = "Python"
for letter in word:
    print(letter)

# Lặp qua dictionary
student = {"name": "Bình", "age": 20, "grade": "A"}
for key, value in student.items():
    print(f"{key}: {value}")

# Lặp với enumerate (lấy cả index và value)
colors = ["đỏ", "xanh", "vàng"]
for index, color in enumerate(colors):
    print(f"Màu thứ {index + 1}: {color}")

Các cách dùng for:

  • for item in list: Lặp qua list
  • range(start, stop, step): Tạo dãy số
  • for char in string: Lặp qua từng ký tự
  • for key, value in dict.items(): Lặp qua dictionary
  • enumerate(): Lấy cả index và giá trị
  • break: Thoát vòng lặp sớm
  • continue: Bỏ qua vòng lặp hiện tại

Vòng lặp while

while loop và ứng dụng

# while cơ bản
count = 0
while count < 5:
    print(f"Đếm: {count}")
    count += 1   # Tăng count lên 1

# while với break
number = 0
while True:
    print(number)
    number += 1
    if number >= 5:
        break   # Thoát vòng lặp

# while với continue
num = 0
while num < 10:
    num += 1
    if num % 2 == 0:   # Nếu số chẵn
        continue  # Bỏ qua, không in
    print(f"Số lẻ: {num}")  # Chỉ in số lẻ

# while-else (thực hiện else khi điều kiện sai)
attempt = 0
max_attempts = 3

while attempt < max_attempts:
    print(f"Thử lần {attempt + 1}")
    attempt += 1
else:
    print("Đã thử đủ số lần!")

# Ứng dụng: đoán số
secret_number = 7
guess = 0

while guess != secret_number:
    guess = int(input("Đoán số (1-10): "))
    if guess < secret_number:
        print("Số quá nhỏ!")
    elif guess > secret_number:
        print("Số quá lớn!")
    else:
        print("Chính xác! Bạn đã đoán đúng!")

Vòng lặp while:

  • Cơ bản: Lặp khi điều kiện còn đúng
  • break: Thoát vòng lặp ngay lập tức
  • continue: Bỏ qua phần còn lại, tiếp tục vòng mới
  • while-else: else chạy khi điều kiện while sai
  • Ứng dụng: Game, menu, xử lý cho đến khi đúng

Hàm (Functions)

Định nghĩa và sử dụng hàm

# Hàm cơ bản không tham số
def greet():
    print("Xin chào!")

# Hàm có tham số
def greet_person(name):
    print(f"Xin chào {name}!")

# Hàm có giá trị trả về
def add(a, b):
    return a + b

# Hàm với tham số mặc định
def power(base, exponent=2):
    return base ** exponent

# Hàm với nhiều giá trị trả về
def calculate(a, b):
    sum = a + b
    difference = a - b
    product = a * b
    quotient = a / b if b != 0 else "Không thể chia cho 0"
    return sum, difference, product, quotient

# Hàm với số lượng tham số không xác định (*args)
def sum_all(*numbers):
    total = 0
    for num in numbers:
        total += num
    return total

# Hàm với từ khóa tham số không xác định (**kwargs)
def print_info(**info):
    for key, value in info.items():
        print(f"{key}: {value}")

# Gọi hàm
greet()
greet_person("Bình")
print("Tổng:", add(5, 3))
print("3^2 =", power(3))
print("2^3 =", power(2, 3))

results = calculate(10, 5)
print(f"Tổng: {results[0]}, Hiệu: {results[1]}, Tích: {results[2]}, Thương: {results[3]}")

print("Tổng nhiều số:", sum_all(1, 2, 3, 4, 5))

print_info(name="Bình", age=25, city="Hà Nội")

Các loại hàm:

  • Hàm không tham số: def func():
  • Hàm có tham số: def func(param):
  • Hàm trả về giá trị: return value
  • Tham số mặc định: def func(param=default):
  • Nhiều giá trị trả về: Trả về tuple
  • *args: Nhận nhiều tham số không tên
  • **kwargs: Nhận nhiều tham số có tên

List (Danh sách)

Tất cả phương thức list

# Tạo list
numbers = [1, 2, 3, 4, 5]
fruits = ["táo", "chuối", "cam"]
mixed = [1, "hello", 3.14, True]

# Truy cập phần tử
print("Phần tử đầu:", fruits[0])      # táo
print("Phần tử cuối:", fruits[-1])    # cam
print("Từ vị trí 1 đến 2:", fruits[1:3]) # ['chuối', 'cam']

# Thay đổi phần tử
fruits[0] = "dâu"
print("Sau khi thay đổi:", fruits)

# Thêm phần tử
fruits.append("nho")          # Thêm vào cuối
fruits.insert(1, "xoài")       # Thêm vào vị trí 1

# Xóa phần tử
fruits.remove("chuối")       # Xóa phần tử "chuối"
popped = fruits.pop()        # Xóa và trả về phần tử cuối
print(f"Đã xóa: {popped}")
del fruits[0]               # Xóa phần tử ở vị trí 0

# Các phương thức khác
numbers2 = [6, 7, 8]
numbers.extend(numbers2)     # Mở rộng list
print("Số phần tử:", len(numbers))  # Độ dài list
print("Tổng:", sum(numbers))       # Tổng các số
print("Max:", max(numbers))        # Giá trị lớn nhất
print("Min:", min(numbers))        # Giá trị nhỏ nhất

# Sắp xếp
unsorted = [3, 1, 4, 1, 5, 9, 2]
unsorted.sort()                 # Sắp xếp tăng dần
print("Tăng dần:", unsorted)
unsorted.sort(reverse=True)   # Sắp xếp giảm dần
print("Giảm dần:", unsorted)

# List comprehension (tạo list nhanh)
squares = [x**2 for x in range(10)]
print("Bình phương từ 0-9:", squares)

even_numbers = [x for x in range(20) if x % 2 == 0]
print("Số chẵn từ 0-19:", even_numbers)

Phương thức List:

  • append(): Thêm vào cuối
  • insert(): Thêm vào vị trí chỉ định
  • remove(): Xóa phần tử cụ thể
  • pop(): Xóa và trả về phần tử
  • extend(): Mở rộng list
  • sort(): Sắp xếp
  • reverse(): Đảo ngược
  • len(): Độ dài list
  • List comprehension: Tạo list nhanh [x for x in range()]

Dictionary (Từ điển)

Key-value pairs

# Tạo dictionary
student = {
    "name": "Nguyễn Văn Bình",
    "age": 20,
    "grades": [8.5, 9.0, 7.5],
    "is_student": True
}

# Truy cập giá trị
print("Tên:", student["name"])
print("Tuổi:", student.get("age"))
print("Điểm:", student.get("grades", "Không có điểm"))

# Thêm/Sửa giá trị
student["major"] = "Computer Science"  # Thêm mới
student["age"] = 21                         # Sửa giá trị

# Xóa phần tử
removed = student.pop("is_student")     # Xóa và trả về giá trị
del student["grades"]                   # Xóa key

# Kiểm tra key tồn tại
if "name" in student:
    print("Tên tồn tại trong dictionary")

# Lặp qua dictionary
for key in student:
    print(f"{key}: {student[key]}")

for key, value in student.items():
    print(f"{key}: {value}")

# Các phương thức
print("Tất cả keys:", list(student.keys()))
print("Tất cả values:", list(student.values()))
print("Tất cả items:", list(student.items()))

# Dictionary comprehension
squares = {x: x**2 for x in range(1, 6)}
print("Bảng bình phương:", squares)

# Dictionary lồng nhau
school = {
    "student1": {"name": "Bình", "age": 20},
    "student2": {"name": "An", "age": 21},
    "student3": {"name": "Minh", "age": 19}
}
print("Student2:", school["student2"]["name"])

Dictionary methods:

  • get(key, default): Lấy giá trị, trả về default nếu không có
  • keys(): Trả về tất cả keys
  • values(): Trả về tất cả values
  • items(): Trả về cặp (key, value)
  • pop(key): Xóa và trả về giá trị
  • update(): Cập nhật dictionary
  • clear(): Xóa tất cả phần tử
  • Dict comprehension: {key: value for item in iterable}

  • "b": Chế độ binary

Phương thức:

Error Handling

Xử lý lỗi try-except

# try-except cơ bản
try:
    number = int(input("Nhập số: "))
    print(f"Số bạn nhập: {number}")
except ValueError:
    print("Lỗi: Bạn phải nhập số nguyên!")

# Nhiều except blocks
try:
    a = 10
    b = 0
    result = a / b
    print(f"Kết quả: {result}")
except ZeroDivisionError:
    print("Lỗi: Không thể chia cho 0!")
except TypeError:
    print("Lỗi: Kiểu dữ liệu không phù hợp!")
except Exception as e:
    print(f"Lỗi không xác định: {e}")

# try-except-else (không lỗi thì chạy else)
try:
    num = int(input("Nhập số: "))
except ValueError:
    print("Không phải số hợp lệ!")
else:
    print(f"Số hợp lệ: {num}")
    print(f"Bình phương: {num ** 2}")

# try-except-finally (luôn chạy finally)
try:
    file = open("test.txt", "r")
    content = file.read()
    print(content)
except FileNotFoundError:
    print("File không tồn tại!")
finally:
    print("Luôn chạy phần này")
    try:
        file.close()
    except:
        pass

# Tạo lỗi tùy chỉnh (raise)
def check_age(age):
    if age < 0:
        raise ValueError("Tuổi không thể âm!")
    elif age < 18:
        raise ValueError("Bạn chưa đủ 18 tuổi!")
    else:
        return f"Tuổi hợp lệ: {age}"

try:
    print(check_age(25))
    print(check_age(-5))
except ValueError as e:
    print(f"Lỗi: {e}")

Các loại lỗi phổ biến:

  • ValueError: Giá trị không hợp lệ
  • TypeError: Kiểu dữ liệu không phù hợp
  • ZeroDivisionError: Chia cho 0
  • IndexError: Index ngoài phạm vi
  • KeyError: Key không tồn tại trong dict
  • FileNotFoundError: File không tồn tại

Cú pháp:

  • try: Thử chạy code có thể lỗi
  • except Exception: Bắt lỗi cụ thể
  • else: Chạy nếu không có lỗi
  • finally: Luôn chạy dù có lỗi hay không
  • raise Exception(): Tạo lỗi tùy chỉnh

<!-- Geolocation API -->
<button onclick="getLocation()">Lấy vị trí</button>
<p id="location"></p>

<script>
function getLocation() {
    if (navigator.geolocation) {
        navigator.geolocation.getCurrentPosition(showPosition);
    } else {
        document.getElementById("location").innerHTML = 
            "Trình duyệt không hỗ trợ Geolocation.";
    }
}

function showPosition(position) {
    const lat = position.coords.latitude;
    const lon = position.coords.longitude;
    document.getElementById("location").innerHTML = 
        `Vĩ độ: ${lat}<br>Kinh độ: ${lon}`;
}
</script>

<!-- Local Storage -->
<input type="text" id="storageInput" placeholder="Nhập dữ liệu">
<button onclick="saveToStorage()">Lưu</button>
<button onclick="loadFromStorage()">Tải</button>
<button onclick="clearStorage()">Xóa</button>

<script>
function saveToStorage() {
    const data = document.getElementById('storageInput').value;
    localStorage.setItem('userData', data);
    alert('Đã lưu!');
}

function loadFromStorage() {
    const data = localStorage.getItem('userData');
    document.getElementById('storageInput').value = data || '';
}

function clearStorage() {
    localStorage.clear();
    alert('Đã xóa!');
}
</script>

const dropArea = document.getElementById('dropArea');

dropArea.addEventListener('dragover', (e) => {

});

dropArea.addEventListener('drop', (e) => {
    e.preventDefault();

</script>

<!-- Web Workers -->
<button onclick="startWorker()">Start Worker</button>
<button onclick="stopWorker()">Stop Worker</button>
<p id="workerResult"></p>

<script>
let worker;

function startWorker() {
    if (typeof(Worker) !== "undefined") {
        if (!worker) {
            worker = new Worker("worker.js");
        }
        worker.onmessage = function(event) {
            document.getElementById("workerResult").innerHTML = event.data;
        };
    }
}

function stopWorker() {
    if (worker) {
        worker.terminate();
        worker = undefined;
    }
}
</script>

<script>
async function copyToClipboard() {
    const text = document.getElementById('copyText').value;
    try {
        await navigator.clipboard.writeText(text);
        alert('Đã copy!');
    } catch (err) {
        console.error('Copy failed:', err);
    }
}
</script>
